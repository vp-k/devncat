  36: # 안전한 jq 인플레이스 업데이트 (temp 파일 자동 정리)
  37: jq_inplace() {
  38:   local file="$1"; shift
  39:   local tmp
  40:   tmp=$(mktemp)
  41:   if jq "$@" "$file" > "$tmp"; then
  42:     mv "$tmp" "$file"
  43:   else
  44:     rm -f "$tmp"
  45:     die "jq update failed for $file"
  46:   fi
  47: }

  49: # ─── schemaVersion 마이그레이션 (idempotent) ───

  51: # full-auto progress 파일을 v1 → v2로 마이그레이션
  52: # 여러 번 실행해도 안전 (idempotent)
  53: migrate_schema_v2() {
  54:   local file="$1"
  55:   [[ -f "$file" ]] || return 0
  56: 
  57:   # schemaVersion이 이미 2 이상이면 스킵
  58:   local current_ver
  59:   current_ver=$(jq '.schemaVersion // 1' "$file" 2>/dev/null || echo "1")
  60:   [[ "$current_ver" -ge 2 ]] && return 0
  61: 
  62:   # full-auto progress 파일인지 확인 (steps 배열에 phase_0가 있는지)
  63:   local is_full_auto
  64:   is_full_auto=$(jq 'if .steps then [.steps[].name] | any(. == "phase_0") else false end' "$file" 2>/dev/null || echo "false")
  65:   [[ "$is_full_auto" == "true" ]] || return 0
  66: 
  67:   echo "Migrating $file to schemaVersion 2..."
  68:   jq_inplace "$file" '
  69:     .schemaVersion = 2
  70:     | .phases.phase_0.outputs.assumptions //= []
  71:     | .phases.phase_0.outputs.nsm //= null
  72:     | .phases.phase_0.outputs.successCriteria //= []
  73:     | .phases.phase_0.outputs.premortem //= {"tigers":[],"paperTigers":[],"elephants":[]}
  74:     | .phases.phase_0.outputs.projectSize //= null
  75:     | .phases.phase_0.outputs.stakeholders //= null
  76:     | .dod.assumptions_documented //= {"checked":false,"evidence":null}
  77:     | .dod.premortem_done //= {"checked":false,"evidence":null}
  78:     | .dod.launch_ready //= {"checked":false,"evidence":null}
  79:   '
  80:   echo "OK: $file migrated to schemaVersion 2"
  81: }

  94: # --progress-file 인수 파싱 (글로벌)
  95: PROGRESS_FILE=""
  96: parse_progress_file_arg() {
  97:   local args=()
  98:   while [[ $# -gt 0 ]]; do
  99:     case "$1" in
 100:       --progress-file)
 101:         PROGRESS_FILE="${2:?--progress-file requires a path}"
 102:         shift 2
 103:         ;;
 104:       *)
 105:         args+=("$1")
 106:         shift
 107:         ;;
 108:     esac
 109:   done
 110:   # PROGRESS_FILE이 설정되지 않았으면 자동 탐지
 111:   if [[ -z "$PROGRESS_FILE" ]]; then
 112:     PROGRESS_FILE=$(detect_progress_file) || true
 113:   fi
 114:   # 나머지 인수를 REMAINING_ARGS에 저장
 115:   REMAINING_ARGS=("${args[@]+"${args[@]}"}")
 116: }

 437: cmd_init_ralph() {
 438:   local promise="${1:?Usage: init-ralph <promise> <progress_file> [max_iter]}"
 439:   local progress_file="${2:?Usage: init-ralph <promise> <progress_file> [max_iter]}"
 440:   local max_iter="${3:-0}"
 441: 
 442:   mkdir -p .claude
 443: 
 444:   local ralph_file=".claude/ralph-loop.local.md"
 445: 
 446:   if [[ -f "$ralph_file" ]]; then
 447:     echo "WARNING: $ralph_file already exists. Skipping."
 448:     return 0
 449:   fi
 450: 
 451:   local now
 452:   now=$(timestamp)
 453: 
 454:   cat > "$ralph_file" <<ENDRALPH
 455: ---
 456: active: true
 457: iteration: 1
 458: max_iterations: $max_iter
 459: completion_promise: "$promise"
 460: progress_file: "$progress_file"
 461: started_at: "$now"
 462: ---
 463: 
 464: 이전 작업을 이어서 진행합니다.
 465: \`$progress_file\`을 읽고 상태를 확인하세요.
 466: 특히 \`handoff\` 필드를 먼저 읽어 이전 iteration의 맥락을 복구하세요.
 467: 
 468: 1. completed 단계는 건너뛰세요
 469: 2. in_progress 단계가 있으면 해당 단계부터 재개
 470: 3. pending 단계가 있으면 다음 pending 단계 시작
 471: 4. 모든 단계가 completed이고 검증을 통과하면 <promise>$promise</promise> 출력
 472: 
 473: 검증 규칙:
 474: - $progress_file의 모든 단계/문서 status가 completed여야 함
 475: - dod 체크리스트가 모두 checked여야 함
 476: - 조건 미충족 시 절대 <promise> 태그를 출력하지 마세요
 477: ENDRALPH
 478: 
 479:   echo "OK: $ralph_file created (promise: $promise, progress: $progress_file, max_iter: $max_iter)"
 480: }

 482: # ─── status: 현재 상태 요약 출력 ───

 484: cmd_status() {
 485:   require_jq
 486:   require_progress
 487: 
 488:   # schemaVersion 마이그레이션 트리거
 489:   migrate_schema_v2 "$PROGRESS_FILE"
 490: 
 491:   echo "=== Progress Status ($PROGRESS_FILE) ==="
 492: 
 493:   # steps 배열이 있는 경우
 494:   local has_steps
 495:   has_steps=$(jq 'has("steps")' "$PROGRESS_FILE")
 496:   if [[ "$has_steps" == "true" ]]; then
 497:     # 현재 단계 (currentPhase 또는 currentStep 사용)
 498:     local current
 499:     current=$(jq -r '.currentPhase // .currentStep // "unknown"' "$PROGRESS_FILE")
 500:     echo "Current: $current"
 501: 
 502:     # 완료된 단계
 503:     local completed
 504:     completed=$(jq -r '[.steps[] | select(.status == "completed") | (.label // .name)] | join(", ")' "$PROGRESS_FILE")
 505:     [[ -n "$completed" ]] && echo "Completed: $completed"
 506: 
 507:     # 진행 중인 단계
 508:     local in_progress
 509:     in_progress=$(jq -r '[.steps[] | select(.status == "in_progress") | (.label // .name)] | join(", ")' "$PROGRESS_FILE")
 510:     [[ -n "$in_progress" ]] && echo "In Progress: $in_progress"
 511: 
 512:     # 대기 중인 단계
 513:     local pending_count

 580: # ─── update-step: 단계 상태 전이 (동적 검증) ───

 582: cmd_update_step() {
 583:   local step_name="${1:?Usage: update-step <step_name> <status>}"
 584:   local new_status="${2:?Usage: update-step <step_name> <status>}"
 585: 
 586:   require_jq
 587:   require_progress
 588: 
 589:   # schemaVersion 마이그레이션 트리거
 590:   migrate_schema_v2 "$PROGRESS_FILE"
 591: 
 592:   # 유효한 상태 값 확인
 593:   local valid_statuses="pending in_progress completed"
 594:   echo "$valid_statuses" | grep -qw "$new_status" || die "Invalid status: $new_status. Valid: $valid_statuses"
 595: 
 596:   # progress 파일에서 해당 step이 존재하는지 동적으로 확인
 597:   local step_exists
 598:   step_exists=$(jq --arg name "$step_name" '[.steps[] | select(.name == $name)] | length' "$PROGRESS_FILE")
 599:   [[ "$step_exists" -gt 0 ]] || die "Step not found: $step_name. Available steps: $(jq -r '[.steps[].name] | join(", ")' "$PROGRESS_FILE")"
 600: 
 601:   # Pre-mortem 하드 게이트: phase_2 진입 시 blocking Tiger 미해결 검사
 602:   if [[ "$step_name" == "phase_2" && "$new_status" == "in_progress" ]]; then
 603:     local blocking_unresolved
 604:     blocking_unresolved=$(jq '
 605:       [.phases.phase_0.outputs.premortem.tigers // []
 606:        | .[]
 607:        | select(.blocking == true and (.mitigation == null or .mitigation == "" or (.mitigation | test("^\\s*$"))))]
 608:       | length
 609:     ' "$PROGRESS_FILE" 2>/dev/null || echo "0")
 610: 
 611:     if [[ "$blocking_unresolved" -gt 0 ]]; then
 612:       echo "BLOCKED: $blocking_unresolved blocking Tiger(s) have no mitigation."
 613:       echo "Resolve all blocking Tigers before entering Phase 2."
 614:       jq -r '.phases.phase_0.outputs.premortem.tigers // [] | .[] | select(.blocking == true and (.mitigation == null or .mitigation == "" or (.mitigation | test("^\\s*$")))) | "  - \(.risk)"' "$PROGRESS_FILE"
 615:       exit 1
 616:     fi
 617:   fi
 618: 
 619:   # steps 배열에서 해당 step 상태 업데이트 + top-level 갱신
 620:   jq_inplace "$PROGRESS_FILE" --arg name "$step_name" --arg status "$new_status" '
 621:     (.steps[] | select(.name == $name)).status = $status
 622:     | if $status == "in_progress" then
 623:         (if has("currentPhase") then .currentPhase = $name else . end)
 624:         | (if has("currentStep") then .currentStep = $name else . end)
 625:       else . end
 626:     | if has("handoff") and (.handoff | has("currentPhase")) then
 627:         .handoff.currentPhase = (.currentPhase // null)
 628:       else . end
 629:     | .status = (if ([.steps[].status] | all(. == "completed")) then "completed" else "in_progress" end)
 630:   '
 631: 
 632:   echo "OK: $step_name -> $new_status"
 633: }

 863:     --exclude='*.woff2'

 881: $masked_matches

 902:     jq -n --arg ts "$ts" --argjson count "$found" --arg result "$scan_result" \

 904:   fi

 910:     echo "ACTION: Remove secrets and use environment variables instead."

 912:   else
 913:     echo "[secret-scan] PASS (no secrets detected)"
 914:     echo "=== SECRET SCAN PASSED ==="
 915:     return 0
 916:   fi
 917: }

1116: # ─── record-error: 에러 반복 판별 + errorHistory 업데이트 ───

1118: cmd_record_error() {
1119:   local err_file="" err_type="" err_msg="" err_level="" err_action="" err_result="" reset_count=false
1120: 
1121:   while [[ $# -gt 0 ]]; do
1122:     case "$1" in
1123:       --file)        err_file="${2:?--file requires a value}"; shift 2 ;;
1124:       --type)        err_type="${2:?--type requires a value}"; shift 2 ;;
1125:       --msg)         err_msg="${2:?--msg requires a value}"; shift 2 ;;
1126:       --level)       err_level="${2:?--level requires L0-L4}"; shift 2 ;;
1127:       --action)      err_action="${2:?--action requires a description}"; shift 2 ;;
1128:       --result)      err_result="${2:?--result requires pass|fail}"; shift 2 ;;
1129:       --reset-count) reset_count=true; shift ;;
1130:       *)             shift ;;
1131:     esac
1132:   done
1133: 
1134:   [[ -n "$err_file" ]] || die "Usage: record-error --file <f> --type <t> --msg <m> [--level L0-L4] [--action '...'] [--result pass|fail] [--reset-count]"
1135:   [[ -n "$err_type" ]] || die "Usage: record-error --file <f> --type <t> --msg <m>"
1136:   [[ -n "$err_msg" ]]  || die "Usage: record-error --file <f> --type <t> --msg <m>"
1137: 
1138:   require_jq
1139:   require_progress
1140: 
1141:   # 에러 레벨 유효성 검사
1142:   if [[ -n "$err_level" ]]; then
1143:     echo "L0 L1 L2 L3 L4" | grep -qw "$err_level" || die "Invalid level: $err_level. Valid: L0 L1 L2 L3 L4"
1144:   fi
1145: 
1146:   # 에스컬레이션 레벨별 예산
1147:   # L0=3, L1=3, L2=1, L3=3, L4=1
1148:   local -A level_budget=( ["L0"]=3 ["L1"]=3 ["L2"]=1 ["L3"]=3 ["L4"]=1 )
1149: 
1150:   # 현재 errorHistory 읽기
1151:   local current_err_type current_err_file current_count
1152:   current_err_type=$(jq -r '.errorHistory.currentError.type // ""' "$PROGRESS_FILE")
1153:   current_err_file=$(jq -r '.errorHistory.currentError.file // ""' "$PROGRESS_FILE")
1154:   current_count=$(jq '.errorHistory.currentError.count // 0' "$PROGRESS_FILE")
1155: 
1156:   # 현재 에스컬레이션 레벨/예산 읽기
1157:   local current_escalation current_budget
1158:   current_escalation=$(jq -r '.errorHistory.escalationLevel // "L0"' "$PROGRESS_FILE")
1159:   current_budget=$(jq '.errorHistory.escalationBudget // 3' "$PROGRESS_FILE")
1160: 
1161:   # --level이 제공되면 에스컬레이션 레벨/예산 항상 반영
1162:   if [[ -n "$err_level" ]]; then
1163:     current_escalation="$err_level"
1164:     current_budget="${level_budget[$err_level]:-3}"
1165:   fi
1166: 
1167:   # --reset-count 시 카운터 리셋
1168:   if [[ "$reset_count" == "true" ]]; then
1169:     current_count=0
1170:   fi
1171: 
1172:   # 동일 에러 판별 (type + file + 메시지 핵심 일치)
1173:   # 메시지 정규화: 숫자/라인번호 제거하여 핵심만 비교
1174:   local msg_normalized
1175:   msg_normalized=$(echo "$err_msg" | sed 's/[0-9]//g' | sed 's/  */ /g' | head -c 100)
1176:   local prev_msg_normalized
1177:   prev_msg_normalized=$(jq -r '.errorHistory.currentError.msgNormalized // ""' "$PROGRESS_FILE" 2>/dev/null)
1178:   if [[ "$current_err_type" == "$err_type" ]] && [[ "$current_err_file" == "$err_file" ]] && [[ "$msg_normalized" == "$prev_msg_normalized" ]]; then
1179:     current_count=$((current_count + 1))
1180:   else
1181:     current_count=1
1182:   fi
1183: 
1184:   # 진행/회귀 판별 (에러 레벨 기반)
1185:   local direction="same"
1186:   if [[ -n "$err_level" ]]; then
1187:     local level_history
1188:     level_history=$(jq -r '.errorHistory.levelHistory // [] | .[-1] // ""' "$PROGRESS_FILE")
1189:     if [[ -n "$level_history" ]] && [[ "$level_history" != "$err_level" ]]; then
1190:       local prev_num=${level_history#L}
1191:       local curr_num=${err_level#L}
1192:       if [[ "$curr_num" -gt "$prev_num" ]]; then
1193:         direction="forward"
1194:       elif [[ "$curr_num" -lt "$prev_num" ]]; then
1195:         direction="backward"
1196:       fi
1197:     fi
1198: 
1199:     # 회귀 연속 횟수 체크
1200:     if [[ "$direction" == "backward" ]]; then
1201:       local last_two_directions
1202:       last_two_directions=$(jq -r '
1203:         .errorHistory.levelHistory // [] |
1204:         if length >= 2 then
1205:           [.[length-2], .[length-1]] |
1206:           if .[0] > .[1] then "backward" else "not" end
1207:         else "not" end
1208:       ' "$PROGRESS_FILE")
1209:       if [[ "$last_two_directions" == "backward" ]]; then
1210:         echo "WARNING: 회귀 2회 연속 — 현재 접근법을 재검토하세요 (codex 호출 또는 다른 접근법)"
1211:       fi
1212:     fi
1213:   fi
1214: 
1215:   # 에스컬레이션 로그 엔트리 생성
1216:   local ts
1217:   ts=$(timestamp)
1218:   local log_entry
1219:   log_entry=$(jq -n \
1220:     --arg ts "$ts" \
1221:     --arg level "${err_level:-$current_escalation}" \
1222:     --argjson attempt "$current_count" \
1223:     --arg error "$err_msg" \
1224:     --arg action "${err_action:-}" \
1225:     --arg result "${err_result:-fail}" \
1226:     '{ts: $ts, level: $level, attempt: $attempt, error: $error, action: $action, result: $result}')
1227: 
1228:   # errorHistory 업데이트 (확장된 구조)
1229:   jq_inplace "$PROGRESS_FILE" \
1230:     --arg type "$err_type" \
1231:     --arg file "$err_file" \
1232:     --arg msg "$err_msg" \
1233:     --argjson count "$current_count" \
1234:     --arg escalation "$current_escalation" \
1235:     --argjson budget "$current_budget" \
1236:     --arg level "${err_level:-}" \
1237:     --arg mnorm "$msg_normalized" \
1238:     --argjson logEntry "$log_entry" '
1239:     .errorHistory.currentError = {
1240:       "type": $type,
1241:       "file": $file,
1242:       "message": $msg,
1243:       "msgNormalized": $mnorm,
1244:       "count": $count,
1245:       "escalationLevel": $escalation
1246:     }
1247:     | .errorHistory.attempts += [$msg]
1248:     | .errorHistory.escalationLevel = $escalation
1249:     | .errorHistory.escalationBudget = $budget
1250:     | if $level != "" then
1251:         .errorHistory.levelHistory = ((.errorHistory.levelHistory // []) + [$level])
1252:       else . end
1253:     | .errorHistory.escalationLog = ((.errorHistory.escalationLog // []) + [$logEntry])
1254:   '
1255: 
1256:   echo "Error recorded: $err_type in $err_file (count: $current_count, escalation: $current_escalation)"
1257:   [[ -n "$err_level" ]] && echo "DIRECTION: $direction (error level: $err_level)"
1258: 
1259:   # exit code로 에스컬레이션 결과 전달
1260:   # exit 0: 현재 레벨 예산 내 → 계속 시도
1261:   # exit 1: 현재 레벨 예산 소진 → 다음 레벨로 에스컬레이트
1262:   # exit 2: L2 도달 → codex 분석 필요
1263:   # exit 3: L5 도달 → 사용자 개입 필요
1264:   if [[ "$current_escalation" == "L4" ]] && [[ $current_count -ge ${level_budget[L4]} ]]; then
1265:     echo "ACTION: L4 예산 소진 → L5 사용자 개입 필요"
1266:     exit 3
1267:   elif [[ "$current_escalation" == "L2" ]]; then
1268:     echo "ACTION: L2 → codex 분석 필요"
1269:     exit 2
1270:   elif [[ $current_count -ge $current_budget ]]; then
1271:     echo "ACTION: $current_escalation 예산 소진 ($current_count/$current_budget) → 다음 레벨로 에스컬레이트"
1272:     exit 1
1273:   else
1274:     echo "ACTION: 계속 시도 ($current_count/$current_budget)"
1275:     exit 0
1276:   fi
1277: }

1430:   else

1432:   fi

1435:   echo ""
